1//! DiffusionX is a multi-threaded high-performance Rust library for random number/stochastic process simulation.
2//! # Usage
3//! ```rust
4//! use diffusionx::simulation::{Bm, Simulation, Functional};
5//! // Brownian motion simulation
6//! let bm = Bm::new(0.0, 1.0, 1.0)?;  // Create Brownian motion object: initial position 0, diffusion coefficient 1, duration 1
7//! let time_step = 0.01;  // Time step
8//! let (times, positions) = bm.simulate(time_step)?;  // Simulate Brownian motion trajectory
9//! // Monte Carlo simulation of Brownian motion statistics
10//! let mean = bm.mean(time_step, 1000)?;  // Mean  bm.raw_moment(time_step, 1, 1000)?;
11//! let msd = bm.msd(time_step, 1000)?;  // Mean square displacement  bm.central_moment(time_step, 2, 1000)?;
12//! // First passage time of Brownian motion
13//! let max_duration = 1000; // if over this duration, the simulation will be terminated and return None
14//! let fpt = bm.fpt(time_step, (-1.0, 1.0), max_duration)?;
15//! ```
16//! # Progress
17//! ## Random Number Generation
18//! - [x] Normal distribution
19//! - [x] Uniform distribution
20//! - [x] Exponential distribution
21//! - [x] Poisson distribution
22//! - [x] Alpha-stable distribution
23//! ## Stochastic Processes
24//! - [x] Brownian motion
25//! - [ ] Alpha-stable Lévy process
26//! - [ ] Fractional Brownian motion
27//! - [ ] Poisson process
28//! - [ ] Compound Poisson process
29//! - [ ] Langevin equation
