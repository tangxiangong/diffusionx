5use rand::distr::uniform::Error as UniformError;
6use rand_distr::{ExpError, NormalError, PoissonError};
7use thiserror::Error;
8
9/// Result type for the crate
10pub type XResult<T> = Result<T, XError>;
11
12/// Error type for the crate
13///
14/// This enum represents the errors that can occur in the crate.
15///
16#[derive(Error, Debug, PartialEq, Eq, Clone)]
17pub enum XError {
18    #[error("Sample Uniform Distribution Error: {0}")]
19    UniformSampleError(#[from] UniformError),
20    #[error("Sample Normal Distribution Error: {0}")]
21    NormalSampleError(#[from] NormalError),
22    #[error("Sample Poisson Distribution Error: {0}")]
23    PoissonSampleError(#[from] PoissonError),
24    #[error("Sample Exponential Distribution Error: {0}")]
25    ExpSampleError(#[from] ExpError),
26    #[error("Sample Stable Distribution Error: {0}")]
27    StableSampleError(#[from] StableError),
28    #[error("Probability must be between 0 and 1")]
29    BoolSampleError,
30    #[error("Simulate Error: {0}")]
31    SimulateError(#[from] SimulationError),
32}
33
34#[derive(Error, Debug, PartialEq, Eq, Clone, Copy)]
35pub enum StableError {
36    #[error("Index of stability must be in the range (0, 2]")]
37    InvalidIndex,
38    #[error("Skewness parameter must be in the range [-1, 1]")]
39    InvalidSkewness,
40    #[error("Scale parameter must be positive")]
41    InvalidScale,
42    #[error("Location parameter must be a real number")]
43    InvalidLocation,
44    #[error("Index of stability must be in the range (0, 1)")]
45    InvalidSkewIndex,
46}
47
48#[derive(Error, Debug, PartialEq, Eq, Clone)]
49pub enum SimulationError {
50    #[error("Invalid parameters: {0}")]
51    InvalidParameters(String),
52    #[error("Invalid time step: {0}")]
53    InvalidTimeStep(String),
54    #[error("Invalid time interval: {0}")]
55    InvalidTimeInterval(String),
56    #[error("Unknown error, simulation failed")]
57    Unknown,
58}
